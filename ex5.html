Binding de clases y estilos

            
<div id="forbidden-name">
    <template v-if="name">
        <h4>Hola, {{ name }}</h4>
        <p>Bienvenido a Hogwarts</p>
    </template>

    <p v-else>Por favor escribe tu nombre:</p>

    <form v-on:submit="signUp($event)">

        <input type="text" v-model="name" v-on:keyup="checkName()" class="form-control">

        <p>¿A qué casa te gustaría pertenecer?</p>

        <label v-bind:class="{ hide : nodebesernombrado }">
            Gryffindor: <input type="radio" v-model="house" value="Gryffindor">
        </label>

        <label v-bind:class="{ hide : nodebesernombrado }">
            Hufflepuff: <input type="radio" v-model="house" value="Hufflepuff">
        </label>

        <label v-bind:class="{ hide : nodebesernombrado }">
            Ravenclaw: <input type="radio" v-model="house" value="Ravenclaw">
        </label>

        <label>
            Slytherin: <input type="radio" v-model="house" value="Slytherin">
        </label>

        <p v-show="name">
            <button type="submit" class="button">Regístrate</button>
        </p>
    </form>

    <pre>
        <code class="language-xml">{{ $data }}</code>
    </pre>
</div>
            
        
            
new Vue({
    el: "#forbidden-name",
    data: {
        name: "",
        house: "",
        nodebesernombrado : false
    },

    methods: {
        signUp: function (event) {

            event.preventDefault();

            console.log("Enhorabuena " + this.name + " te has registrado!!! Has elegido pertenecer a " + this.house+"!!!");

        },

        checkName: function () {
            if (this.name == "Voldemort") {
                this.nodebesernombrado = true;
            } else {
                this.nodebesernombrado = false;
            }
        }
    }
});
